55template<class T>
56struct Polarized {
57 T vert, horz;
58 Polarized() {}
59 Polarized(const Polarized<T>& p) : vert(p.vert), horz(p.horz) {}
60 Polarized(const T& v, const T& h) : vert(v), horz(h) {}
61 Polarized(T&& v, T&& h) : vert(std::move(v)), horz(std::move(h)) {}
62 T& operator[] (unsigned n) {
63 assert(n < 2);

82template<class T>
83bool eq_dims(const Vector3d<T>& a) {
84 const uint d1 = a[0].size();
85 const uint d2 = a[0][0].size();
86 // проверяем, что a - на самом деле a[x][d1][d2]
87 for (uint i = 1; i < a.size(); i++) {
88 if (a[i].size() != d1)
89 return false;
90 for (uint j = 1; j < d1; j++)
91 if (a[i][j].size() != d2)
92 return false;
93 }
94 return true;
95}

99template<class T>
100bool eq_dims(const Vector3d<T>& a, const Vector3d<T>& b) {
101 return
102 eq_dims(a)
103 && eq_dims(b)
104 && a.size() == b.size()
105 && a[0].size() == b[0].size()
106 && a[0][0].size() == b[0][0].size();
107}

111bool eq_dims(const Polarized<T>& a) {
