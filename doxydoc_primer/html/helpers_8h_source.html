101// Реализация выражения (int32_t)round((float)x / pow(2, n)) без использования "math" функций
102static inline int32_t round_i32(int32_t x, const unsigned n)
103{
104 if (x == 0 || n == 0)
105 return x;
106 const int_least8_t s = (x < 0) ? -1 : 1;
107 x *= s;
108 // IEEE 754 32-bit float
109 //
110 // 1 bit - sign
111 // 8 bits - exponent
112 // 23 bits - fraction (22 stored)
113 //
114 // S.EEEEEEEE._FFFFFFFFFFFFFFFFFFFFFF
115 //
116 // Если "ширина" 'x' более 22-bit (fraction) следует выполнить округление
117 // с помощью приведения типа к float и обратно: x = (int32_t)((float)x)
118 if (x & 0x7FC00000) // EEEEEEEEE._
119 x = (float)x;
120 return round_u32(s, x, n);
121}

123// Реализация выражения (int64_t)round((double)x / pow(2, n)) без использования "math" функций
124static inline int64_t round_i64(int64_t x, const unsigned n)
125{
126 if (x == 0 || n == 0)
127 return x;
128 const int_least8_t s = (x < 0) ? -1 : 1;
129 x *= s;
130 // IEEE 754 64-bit float (double)
131 //
132 // 1 bit - sign
133 // 11 bits - exponent
134 // 53 bits - fraction (52 stored)
135 //
136 // S.EEEEEEEEEEE._FFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFF
137 //
138 // Если "ширина" 'x' более 52-bit (fraction) следует выполнить округление
139 // с помощью приведения типа к double и обратно: x = (int64_t)((double)x)
140 if (x & 0x7FF8000000000000) // EEEEEEEEEEE._
141 x = (double)x;
142 return round_u64(s, x, n);
143}

145// Внимание: аргумент 'x' д. быть положительным целым в диапазоне: 0...INT32_MAX
146static inline int32_t round_u32(const int s, const uint32_t x, const unsigned n)
147{
148 const uint32_t y = (uint32_t) 1 << n;
149 const uint32_t y2 = y >> 1; // y/2
150 const uint32_t rem = x & (y - 1);
151 const uint32_t add = (rem >= y2) ? y : 0;
152 return s * ((x - rem + add) >> n);
153}

155// Внимание: аргумент 'x' д. быть положительным целым в диапазоне: 0...INT64_MAX
156static inline int64_t round_u64(const int s, const uint64_t x, const unsigned n)
157{
158 const uint64_t y = (uint64_t) 1 << n;
159 const uint64_t y2 = y >> 1; // y/2
160 const uint64_t rem = x & (y - 1);
161 const uint64_t add = (rem >= y2) ? y : 0;
162 return s * ((x - rem + add) >> n);
163}
