312 {
313 // adapted from http://www.gnuplot.info/files/gpReadMouseTest.c
314 if(0 > openpty(&master_fd, &slave_fd, NULL, NULL, NULL)) {
315 perror("openpty");
316 throw std::runtime_error("openpty failed");
317 }

319 if(ttyname_r(slave_fd, pty_fn_buf, 1024)) {
320 perror("ttyname_r");
321 throw std::runtime_error("ttyname failed");
322 }
323 pty_fn = std::string(pty_fn_buf);
324 if(debug_messages) {
325 std::cerr << "feedback_fn=" << pty_fn << std::endl;
326 }
327
328 // disable echo
329 struct termios tios;
330 if(tcgetattr(slave_fd, &tios) < 0) {
331 perror("tcgetattr");
332 throw std::runtime_error("tcgetattr failed");
333 }
334 tios.c_lflag &= ~(ECHO | ECHONL);
335 if(tcsetattr(slave_fd, TCSAFLUSH, &tios) < 0) {
336 perror("tcsetattr");
337 throw std::runtime_error("tcsetattr failed");
338 }
339
340 pty_fh = fdopen(master_fd, "r");
341 if(!pty_fh) {
342 throw std::runtime_error("fdopen failed");
343 }
344 }
345
346private:
347 // noncopyable
348 GnuplotFeedbackPty(const GnuplotFeedbackPty &);
349 const GnuplotFeedbackPty& operator=(const GnuplotFeedbackPty &);
350
351public:
352 ~GnuplotFeedbackPty() {
353 if(pty_fh) fclose(pty_fh);
354 if(master_fd > 0) ::close(master_fd);
355 if(slave_fd > 0) ::close(slave_fd);
356 }
357
358 std::string filename() const {
359 return pty_fn;
360 }
361
362 FILE *handle() const {
363 return pty_fh;
364 }
365
366private:
367 std::string pty_fn;
368 FILE *pty_fh;
369 int master_fd, slave_fd;

413// {{{1 Traits and printers for entry datatypes
414//
415// This section contains the mechanisms for sending scalar and tuple data to gnuplot. Pairs
416// and tuples are sent by appealing to the senders defined for their component scalar types.
417// Senders for arrays are defined in a later section.
418//
419// There are three classes which need to be specialized for each supported datatype:
420// 1. TextSender to send data as text. The default is to just send using the ostream's `<<`
421// operator.
422// 2. BinarySender to send data as binary, in a format which gnuplot can understand. There is
423// no default implementation (unimplemented types raise a compile time error), however
424// inheriting from FlatBinarySender will send the data literally as it is stored in memory.
425// This suffices for most of the standard built-in types (e.g. uint32_t or double).
426// 3. BinfmtSender sends a description of the data format to gnuplot (e.g. `%uint32`). Type
427// `show datafile binary datasizes` in gnuplot to see a list of supported formats.
428
429// {{{2 Basic entry datatypes
430
431// Default TextSender, sends data using `<<` operator.
432template <typename T, typename Enable=void>
433struct TextSender {
434 static void send(std::ostream &stream, const T &v) {
435 stream << v;
436 }
437};
438
439// Default BinarySender, raises a compile time error.
440template <typename T, typename Enable=void>
441struct BinarySender {
442 GNUPLOT_STATIC_ASSERT_MSG((sizeof(T) == 0), "BinarySender class not specialized for this type");
443
444 // This is here to avoid further compilation errors, beyond what the assert prints.
445 static void send(std::ostream &stream, const T &v);
446};

448// This is a BinarySender implementation that just sends directly from memory. Data types
449// which can be sent this way can have their BinarySender specialization inherit from this.
450template <typename T>
451struct FlatBinarySender {
452 static void send(std::ostream &stream, const T &v) {
453 stream.write(reinterpret_cast<const char *>(&v), sizeof(T));
454 }
455};

457// Default BinfmtSender, raises a compile time error.
458template <typename T, typename Enable=void>
459struct BinfmtSender {

461
462 // This is here to avoid further compilation errors, beyond what the assert prints.
463 static void send(std::ostream &);
464};

743// {{{1 ArrayTraits and Range classes
744//
745// This section handles sending of array data to gnuplot. It is rather complicated because of
746// the diversity of storage schemes supported. For example, it treats a
747// `std::pair<std::vector<T>, std::vector<U>>` in the same way as a
748// `std::vector<std::pair<T, U>>`, iterating through the two arrays in lockstep, and sending
749// pairs <T,U> to gnuplot as columns. In fact, any nested combination of pairs, tuples, STL
750// containers, Blitz arrays, and Armadillo arrays is supported (with the caveat that, for
751// instance, Blitz arrays should never be put into an STL container or you will suffer
752// unpredictable results due the way Blitz handles assignment). Nested containers are
753// considered to be multidimensional arrays. Although gnuplot only supports 1D and 2D arrays,
754// our module is in principle not limited.
755//
756// The ArrayTraits class is specialized for every supported array or container type (the
757// default, unspecialized, version of ArrayTraits exists only to tell you that something is
758// *not* a container, via the is_container flag). ArrayTraits tells you the depth of a nested
759// (or multidimensional) container, as well as the value type, and provides a specialized
760// sort of iterator (a.k.a. "range"). Ranges are sort of like STL iterators, except that they
761// have built-in knowledge of the end condition so you don't have to carry around both a
762// begin() and an end() iterator like in STL.
763//
764// As an example of how this works, consider a std::pair of std::vectors. Ultimately this gets
765// sent to gnuplot as two columns, so the two vectors need to be iterated in lockstep.
766// The `value_type` of `std::pair<std::vector<T>, std::vector<U>>` is then `std::pair<T, U>`
767// and this is what deferencing the range (iterator) gives. Internally, this range is built
768// out of a pair of ranges (PairOfRange class), the `inc()` (advance to next element)
769// method calls `inc()` on each of the children, and `deref()` calls `deref()` on each child
770// and combines the results to return a `std::pair`. Tuples are handled as nested pairs.
771//
772// In addition to PairOfRange, there is also a VecOfRange class that can be used to treat the
773// outermost part of a nested container as if it were a tuple. Since tuples are printed as
774// columns, this is like treating a multidimensional array as if it were column-major. A
775// VecOfRange is obtained by calling `get_columns_range`. This is used by, for instance,
776// `send1d_colmajor`. The implementation is similar to that of PairOfRange.
777//
778// The range, accessed via `ArrayTraits<T>::get_range`, will be of a different class depending
779// on T, and this is defined by the ArrayTraits specialization for T. It will always have
780// methods `inc()` to advance to the next element and `is_end()` for checking whether one has
781// advanced past the final element. For nested containers, `deref_subiter()` returns a range
782// iterator for the next nesting level. When at the innermost level of nesting, `deref()`
783// returns the value of the entry the iterator points to (a scalar, pair, or tuple).
784// Only one of `deref()` or `deref_subiter()` will be available, depending on whether there are
785// deeper levels of nesting. The typedefs `value_type` and `subiter_type` tell the return
786// types of these two methods.
787//
788// Support for standard C++ and boost containers and tuples of containers is provided in this
789// section. Support for third party packages like Blitz and Armadillo is in a later section.
790

1096class VecOfRange {

1098 friend void deref_and_print(std::ostream &, const VecOfRange<T> &, PrintMode);

1100public:
1101 VecOfRange() { }
1102 explicit VecOfRange(const std::vector<RT> &_rvec) : rvec(_rvec) { }
1103
1104 static const bool is_container = RT::is_container;
1105 // Don't allow colwrap since it's already wrapped.
1106 static const bool allow_auto_unwrap = false;

1123 for(size_t i=0; i<rvec.size(); i++) {
1124 rvec[i].inc();

1308// {{{2 deref_and_print()
1309//
1310// Dereferences and prints the given range (iterator). At this point we are done with treating
1311// containers as blocks (for 2D data) and lines (for 1D or 2D data). Any further levels of
1312// nested containers will at this point be treated as columns.
1313
1314// If arg is not a container, then print it via send_scalar().
1315template <typename T, typename PrintMode>
1316typename boost::disable_if_c<T::is_container>::type
1317deref_and_print(std::ostream &stream, const T &arg, PrintMode) {
1318 const typename T::value_type &v = arg.deref();
1319 send_scalar(stream, v, PrintMode());
1320}
1321
1322// If arg is a container (but not a PairOfRange or VecOfRange, which are handled below) then
1323// treat the contents as columns, iterating over the contents recursively. If outputting in
1324// text mode, put a space between columns.
1325template <typename T, typename PrintMode>
1326typename boost::enable_if_c<T::is_container>::type
1327deref_and_print(std::ostream &stream, const T &arg, PrintMode) {
1328 if(arg.is_end()) throw plotting_empty_container();
1329 typename T::subiter_type subrange = arg.deref_subiter();
1330 if(PrintMode::is_binfmt && subrange.is_end()) throw plotting_empty_container();
1331 if(debug_array_print && PrintMode::is_text) stream << "{";
1332 bool first = true;
1333 while(!subrange.is_end()) {
1334 if(!first && PrintMode::is_text) stream << " ";
1335 first = false;
1336 deref_and_print(stream, subrange, PrintMode());
1337 subrange.inc();
1338 }
1339 if(debug_array_print && PrintMode::is_text) stream << "}";
1340}
1341
1342// PairOfRange is treated as columns. In text mode, put a space between columns.
1343template <typename T, typename U, typename PrintMode>
1344void deref_and_print(std::ostream &stream, const PairOfRange<T, U> &arg, PrintMode) {
1345 deref_and_print(stream, arg.l, PrintMode());
1346 if(PrintMode::is_text) stream << " ";
1347 deref_and_print(stream, arg.r, PrintMode());
1348}
1349
1350// VecOfRange is treated as columns. In text mode, put a space between columns.
1351template <typename T, typename PrintMode>
1352void deref_and_print(std::ostream &stream, const VecOfRange<T> &arg, PrintMode) {
1353 if(PrintMode::is_binfmt && arg.rvec.empty()) throw plotting_empty_container();
1354 for(size_t i=0; i<arg.rvec.size(); i++) {
1355 if(i && PrintMode::is_text) stream << " ";
1356 deref_and_print(stream, arg.rvec[i], PrintMode());
1357 }
1358}

1362// {{{2 print_block()
1363//
1364// Here we format nested containers into blocks (for 2D data) and lines. Actually, block and
1365// line formatting is only truely needed for text mode output, but for uniformity this function
1366// is also invoked in binary mode (the PrintMode tag determines the output mode). If the goal
1367// is to just print the array size or the binary format string, then the loops exit after the
1368// first iteration.
1369//
1370// The Depth argument tells how deep to recurse. It will be either `2` for 2D data, formatted
1371// into blocks and lines, with empty lines between blocks, or `1` for 1D data formatted into
1372// lines but not blocks. Gnuplot only supports 1D and 2D data, but if it were to support 3D in
1373// the future (e.g. volume rendering), all that would be needed would be some trivial changes
1374// in this section. After Depth number of nested containers have been recursed into, control
1375// is passed to deref_and_print(), which treats any further nested containers as columns.
1376
1377// Depth==1 and we are not asked to print the size of the array. Send each element of the
1378// range to deref_and_print() for further processing into columns.
1379template <size_t Depth, typename T, typename PrintMode>
1380typename boost::enable_if_c<(Depth==1) && !PrintMode::is_size>::type
1381print_block(std::ostream &stream, T &arg, PrintMode) {
1382 if(PrintMode::is_binfmt && arg.is_end()) throw plotting_empty_container();
1383 for(; !arg.is_end(); arg.inc()) {
1384 //print_entry(arg.deref());
1385 deref_and_print(stream, arg, PrintMode());
1386 // If asked to print the binary format string, only the first element needs to be
1387 // looked at.
1388 if(PrintMode::is_binfmt) break;
1389 if(PrintMode::is_text) stream << std::endl;
1390 }
1391}
1392
1393// Depth>1 and we are not asked to print the size of the array. Loop over the range and
1394// recurse into print_block() with Depth -> Depth-1.
1395template <size_t Depth, typename T, typename PrintMode>
1396typename boost::enable_if_c<(Depth>1) && !PrintMode::is_size>::type
1397print_block(std::ostream &stream, T &arg, PrintMode) {
1398 if(PrintMode::is_binfmt && arg.is_end()) throw plotting_empty_container();
1399 bool first = true;
1400 for(; !arg.is_end(); arg.inc()) {
1401 if(first) {
1402 first = false;
1403 } else {
1404 if(PrintMode::is_text) stream << std::endl;
1405 }
1406 if(debug_array_print && PrintMode::is_text) stream << "<block>" << std::endl;
1407 if(arg.is_end()) throw plotting_empty_container();
1408 typename T::subiter_type sub = arg.deref_subiter();
1409 print_block<Depth-1>(stream, sub, PrintMode());
1410 // If asked to print the binary format string, only the first element needs to be
1411 // looked at.
1412 if(PrintMode::is_binfmt) break;
1413 }
1414}

1416// Determine how many elements are in the given range. Used in the functions below.
1417template <typename T>
1418size_t get_range_size(const T &arg) {
1419 // FIXME - not the fastest way. Implement a size() method for range.
1420 size_t ret = 0;
1421 for(T i=arg; !i.is_end(); i.inc()) ++ret;
1422 return ret;
1423}

1425// Depth==1 and we are asked to print the size of the array.
1426template <size_t Depth, typename T, typename PrintMode>
1427typename boost::enable_if_c<(Depth==1) && PrintMode::is_size>::type
1428print_block(std::ostream &stream, T &arg, PrintMode) {
1429 stream << get_range_size(arg);
1430}

1432// Depth>1 and we are asked to print the size of the array.
1433template <size_t Depth, typename T, typename PrintMode>
1434typename boost::enable_if_c<(Depth>1) && PrintMode::is_size>::type
1435print_block(std::ostream &stream, T &arg, PrintMode) {

1442}
